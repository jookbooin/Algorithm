# << , >> 
`a << b` = a * 2^b
`a >> b` = ( int )a * (1/2)^b

# 비트 마스킹 : 경우의 수, 매개변수
* 불리언 배열 역할 : 1개의 숫자
* 탐색  : 비트 연산자

비트 연산자를 사용하는 이유는 
```
bool[] = { 1, 0, 1, 0}; 과 같은 배열을 
비트로 1010(2)표현해서 `10`으로 대체할 수 있기 때문이다.
```

# 매개 변수 전달
``` cpp
string a[4] = {"사과", "딸기", "포도", "배"};
void go(int n){
    for(int i = 0; i < 4; i++){
        if(n & ( 1 << i)){      // 포함 확인

        }
    }
}

int main(){
    for(int i = 1; i < n; i ++){
        go(1 | ( 1 << i )); // 11, 101, 1001
    }
    return 0;
}
```
